function power(x:integer,n:integer):integer;
Var k,a,b:integer;
Begin
k:=n;a:=1;b:=x;
while k>0 do begin{Niezmiennik:x^n=a*b^k}
if k mod 2 = 0 then begin
k:=k/2;
b:=b*b;
end else begin
k:=k=1;
a:=a*b;
end;
end;
power:=a;
End;

\section{Algorytm}
\begin{algorithm}
\caption{Moj algorytm}\label{alg:cap}
\begin{algorithmic}
\Require $n \geq 0$
\Ensure $a = x^n$
\State $k \gets n; a \gets 1; b \gets x$
\While{$k > 0$} {[Niezmennik: $x^n = a * b^k$]}
\If{$k$ jest liczba parzysta}
    \State $k \gets k / 2$
    \State $b \gets b * b$
\ElsIf{$k$ jest liczba nieparzysta}
    \State $k \gets k - 1$
    \State $a \gets a * b$
\EndIf
\EndWhile
\end{algorithmic}
\end{algorithm}
