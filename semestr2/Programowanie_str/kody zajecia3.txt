unsigned int biggest_proper_divisor(int x){
    int i, temp;
    if (x==0 || x==1) return 1;
    for (i = 1; i < x; i++){
        if(x%i==0){
            temp = i;
        }
    }
    return temp;
}

int main()
{
    int n;
    scanf("%d", &n);
    printf("%d", biggest_proper_divisor(n));
}

#include <stdio.h>
#include <stdlib.h>

unsigned int NWD(unsigned int x, unsigned int y){
    int temp;
    while(x%y!=0){
        temp = x;
        x=y%x;
        y = temp;
    }
    return y;
}
unsigned int suma_wzglednie_pierwszych(unsigned int n){
    int i;
    int suma =0;
    for(i=2;i<n;i++){
        if(NWD(n,i)==1)
            {
                suma=suma+i;
            }
    }
    return suma;
}

int main()
{
    int n;
    scanf("%d", &n);
    printf("%d", suma_wzglednie_pierwszych(n));
}
