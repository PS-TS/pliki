        int fib(int n)
        {
            if(n == 0)
            {
                return 0;
            }
            if(n == 1)
            {
                return 1;
            }

            return fib(n - 1) + fib(n - 2);
        }

        int fib2(int n)
        {
            int[] liczby = new int[n+1];
            liczby[0] = 0;
            liczby[1] = 1;
            for(int i = 2; i <= n; i++)
            {
                liczby[i] = liczby[i - 1] + liczby[i - 2];
            }
            return liczby[n];
        }
        int[] sortowanie(int[] tab)
        {
            int temp;
            bool zamiany = true; 
            
            while (zamiany)  
            {   
                zamiany = false;
                for(int i = 0; i < (tab.Length-1); i++)
                {
                    if (tab[i] > tab[i + 1])
                    {
                        temp = tab[i];
                        tab[i] = tab[i + 1];
                        tab[i + 1] = temp;
                        zamiany = true;
                    }
                }
            }
            
            return tab;
        }

        string ToString(int[] tab)
        {
            string wynik = " ";
            for(int i = 0; i < tab.Length; i++)
            {
                wynik = wynik + tab[i] + ", ";
            }
            return wynik;
        }

        int[] convert(string napis)
        {
            var liczbyS = napis.Trim().Split(' ');
            int[] liczby = new int[liczbyS.Length];
            for(int i =0; i<liczbyS.Length; i++)
            {
                liczby[i] = int.Parse(liczbyS[i]);
            }
            return liczby;
